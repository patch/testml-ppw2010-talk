# t/Bridge.pm for JSON::XS in Perl 5
    
    package t::Bridge;
    use JSON::XS;
    use YAML::XS;
    
    sub to_json   { JSON::XS::encode_json shift->value }
    sub from_json { JSON::XS::decode_json shift->value }
    sub to_yaml   { YAML::XS::Dump        shift->value }
    sub from_yaml { YAML::XS::Load        shift->value }
