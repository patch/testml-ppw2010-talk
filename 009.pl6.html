# t/Bridge.pm for JSON in Perl 6
    
    module t::Bridge;
    use JSON::Tiny;
    use YAML;
    
    sub to_json   ($p) { JSON::Tiny::to-json   $p.value }
    sub from_json ($p) { JSON::Tiny::from-json $p.value }
    sub to_yaml   ($p) { YAML::dump            $p.value }
    sub from_yaml ($p) { YAML::load            $p.value }
